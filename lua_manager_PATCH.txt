    async def seed_spells(self) -> dict:
        """
        Seed the spells table by executing each circle's seed() function.
        Safe to call on every startup — circle modules use ON DUPLICATE KEY UPDATE.
        Returns a { circle_name: count } summary dict.
        """
        from server.core.scripting.loaders.spells_loader import load_spells
        if not self._engine or not self._engine.available:
            log.warning("LuaManager.seed_spells: engine not ready — skipping")
            return {}
        try:
            return load_spells(self._engine) or {}
        except Exception as e:
            log.error("LuaManager.seed_spells failed: %s", e, exc_info=True)
            return {}

    def get_spell(self, spell_number: int) -> Optional[dict]:
        """
        Return a single spell definition dict from the DB, or None.
        Uses Lua's DB bridge so it goes through the same connection pool.
        """
        if not self._engine or not self._engine.available:
            return None
        try:
            result = self._engine.execute(
                f"return require('globals/utils/db')"
                f".queryOne('SELECT * FROM spells WHERE spell_number={int(spell_number)}')"
            )
            if result is None:
                return None
            return self._engine.lua_to_python(result)
        except Exception as e:
            log.error("LuaManager.get_spell(%d): %s", spell_number, e)
            return None
